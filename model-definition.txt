//---------------------------------------------//
// Model definition for Taskforce
//
// Example:
//
// modelname
//      attr1:type1
//      attr2:type2

generation order:   role,       
                    user,       
                    status,     
                    milestone,  
                    component,  
                    priority,   
                    category,   
                    ticket,     
                    comment,    
                    contact     

ticket // defines a task
    id:primary_key
        
    user:references
    status:references
    milestone:references
    component:references
    priority:references
    category:references
    
    feature_request:boolean
    description:text
    reproduction:text
    test_suggestion:text
    conclusion:text
  
user //represents a user
    id:primary_key
    name:string
    email:string
    
    role:references
    
role // viewer, contributor, owner  
    id:primary_key
    name:string
    description:text
    
comment //for a ticket
    id:primary_key
    text:text
    
    ticket:references
    user:references

contact // contact(s) for a ticket
    id:primary_key
    user:references
    ticket:references
    assignee:boolean

status // new, open, feedback, closed
    id:primary_key
    name:string
    description:text

milestone // a time period where tickets can be assigned
    id:primary_key
    due_date:time
    name:string
    description:text
    
component // the corresponding component in the product
    id:primary_key
    name:string
    description:text

priority // the priority (a, b, c or 1, 2, 3 ...)
    id:primary_key
    name:string
    description:text
    
category // the category (major, minor, tweak, trivial, typo, ...)
    id:primary_key
    name:string
    description:text
